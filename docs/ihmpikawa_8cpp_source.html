161 void IhmPikawa::gererEtatPreparation(int etat)
162 {
163  qDebug() << Q_FUNC_INFO << "etat" << etat;
164 
165  changerEcranEtatPreparation();
166  if(etat == EtatPreparation::Repos)
167  {
168  // Café prêt ou au repos
169  afficherPreparationCafePret();
170  decrementerNbCapsules();
171  }
172  else if(etat == EtatPreparation::EnCours)
173  {
174  afficherPreparationCafeEncours();
175  }
176  else if(etat == EtatPreparation::PreparationImpossible)
177  {
178  // Impossible (bac plein, ...)
179  afficherPreparationImpossible();
180  }
181  else if(etat == EtatPreparation::ErreurCapsule)
182  {
183  // Erreur capsule
184  afficherErreurCapsuleAbscente();
185  }
186  else if(etat == EtatPreparation::BacCapsulePlein)
187  {
188  // Erreur reservoir d'eau vide
189  afficherErreurBacCapsulePlein();
190  }
191  else if(etat == EtatPreparation::ReservoirEauPlein)
192  {
193  // Erreur reservoir d'eau vide
194  afficherErreurReservoirEauVide();
195  }
196  else if(etat == EtatPreparation::TasseAbscente)
197  {
198  // Erreur tasse abscente
199  afficherErreurTasseAbscente();
200  }
201  else if(etat == EtatPreparation::CapsuleAbscente)
202  {
203  // Erreur capsule abscente
204  afficherErreurBacCapsulePlein();
205  }
206  else
207  {
208  qDebug() << Q_FUNC_INFO << "Erreur : état inconnu !" << etat;
209  changerEcranAccueil();

326 void IhmPikawa::mettreAJourBarreProgressionPreparationCafe()
327 {
328  // Variables statiques pour conserver l'état entre les appels
329  static int progression = 0;
330  static int dureeTotalePreparation = 0;
331 
332  if(progression == 0)
333  {
334  if(typeDernierCafe == CAFE_RISTRETTO)
335  {
336  dureeTotalePreparation = DUREE_CAFE_COURT;
337  }
338  else if(typeDernierCafe == CAFE_LUNGO)
339  {
340  dureeTotalePreparation = DUREE_CAFE_LONG;
341  }
342  }
343 
344  // Calcule le pas de progression
345  int pasProgression = (100 * DUREE_PROGRESSION) / dureeTotalePreparation;
346 
347  // Mettre à jour la progression
348  progression += pasProgression;
349 
350  // Vérifier si la progression est terminée
351  if(progression >= 100)
352  {
353  minuteurPreparationCafe->stop();
354  ui->progressionCafe->setValue(100);
355  progression = 0; // Réinitialiser la progression pour la prochaine utilisation
356  }
357  else
358  {
359  ui->progressionCafe->setValue(progression);
360  }
361 }
